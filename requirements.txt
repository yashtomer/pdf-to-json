# Direct dependencies only — the libraries this project actually imports.
# Versions are pinned to what was tested. Transitive deps (pdfminer-six,
# pillow, cryptography, etc.) will be resolved automatically by pip.

# pdfplumber: extracts text (and tables) from PDFs that already contain a text
# layer. Used for native-text PDFs like the digitally-signed NICSI Work Orders.
pdfplumber==0.11.9

# pdf2image: converts each PDF page into an image (PNG). Needed because OCR
# engines read images, not PDFs. Used before handing scanned pages to tesseract.
pdf2image==1.17.0

# pytesseract: Python wrapper around the Tesseract OCR engine. Reads text out
# of an image. Used for scanned/image-only PDFs (the VersaLink scan, the
# "Print to PDF" file) that have no extractable text layer.
pytesseract==0.3.13
